import React from 'react'
import { useState } from 'react'

export const Header = (props) => {
    const [show,setShow]=useState(false);
    const Clicked=()=>{
        if (show) {
            setShow(false)
        }
        else{
            setShow(true)
        }
    }
    return (
        <>
        <div className="qna-container">
            <div className="question"><span className="icon" onClick={Clicked}>âž•</span>{props.question}</div>
            {show && <div className="answer">{props.answer}</div>}
        </div>
        </>
    )
}
